cmake_minimum_required(VERSION 3.15)
project(Pathfinding LANGUAGES CXX)

# Standard C++
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# SFML 3
find_package(SFML 3 REQUIRED COMPONENTS system window graphics audio)

file(GLOB SOURCES "src/*.cpp")

add_executable(pathfinding ${SOURCES})

target_link_libraries(pathfinding PRIVATE sfml-system sfml-window sfml-graphics sfml-audio)
